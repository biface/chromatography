1//! Inlet injection profiles — sample description at the column inlet.
2//!
3//! This module defines [`Sample`], which groups the temporal injection profiles
4//! at the column inlet ($z = 0$) for one or more species.
5//!
6//! # Design
7//!
8//! `Sample` uses the same key/value scheme as [`PhysicalModel::set_injections`](crate::physics::PhysicalModel::set_injections):
9//!
10//! | Key | Meaning |
11//! |-----|---------|
12//! | `None` | Default profile — applied to all species without a per-species override |
13//! | `Some(name)` | Per-species override for the named species |
14//!
15//! This scheme is identical to the `HashMap<Option<String>, TemporalInjection>`
16//! already used inside `LangmuirSingle` and `LangmuirMulti`, and integrates
17//! directly with [`PhysicalModel::set_injections`](crate::physics::PhysicalModel::set_injections).
18//!
19//! # Example
20//!
21//! ```rust
22//! use chrom_rs::domain::Sample;
23//! use chrom_rs::models::TemporalInjection;
24//!
25//! // Single Gaussian profile applied to all species
26//! let sample = Sample::uniform(TemporalInjection::gaussian(10.0, 2.0, 0.1));
27//!
28//! // Per-species override
29//! let mut sample = Sample::uniform(TemporalInjection::gaussian(10.0, 2.0, 0.1));
30//! sample.set_species("B", TemporalInjection::rectangle(5.0, 15.0, 0.05));
31//! ```

41/// Inlet injection profiles for a chromatographic run.
42///
43/// Groups [`TemporalInjection`] profiles for one or more species.
44/// The `None` key defines the default profile (all species);
45/// `Some(name)` overrides the profile for a named species.
46///
47/// # Example
48///
49/// ```rust
50/// use chrom_rs::domain::Sample;
51/// use chrom_rs::models::TemporalInjection;
52///
53/// // Single profile for all species
54/// let sample = Sample::uniform(TemporalInjection::dirac(5.0, 0.1));
55/// assert!(sample.default_injection().is_some());
56/// ```
57#[derive(Debug, Clone, Serialize, Deserialize)]
58pub struct Sample {
59    /// Injection profiles indexed by species.
60    ///
61    /// Profils d'injection indexés par espèce :
62    /// - `None`    → profil par défaut pour toutes les espèces
63    /// - `Some(s)` → override pour l'espèce nommée `s`
64    pub injections: HashMap<Option<String>, TemporalInjection>,
65}
66
67impl Sample {
68    // =========================================================================
69    // Constructors
70    // =========================================================================
71
72    /// Creates an empty sample with no injection profiles.
73    ///
74    /// # Example
75    ///
76    /// ```rust
77    /// use chrom_rs::domain::Sample;
78    ///
79    /// let sample = Sample::empty();
80    /// assert!(sample.is_empty());
81    /// ```
82    pub fn empty() -> Self {
83        Self {
84            injections: HashMap::new(),
85        }
86    }
87
88    /// Creates a sample with a single default injection profile applied to
89    /// all species (key `None`).
90    ///
91    /// # Example
92    ///
93    /// ```rust
94    /// use chrom_rs::domain::Sample;
95    /// use chrom_rs::models::TemporalInjection;
96    ///
97    /// let sample = Sample::uniform(TemporalInjection::gaussian(10.0, 2.0, 0.1));
98    /// assert!(sample.default_injection().is_some());
99    /// ```
100    pub fn uniform(injection: TemporalInjection) -> Self {
101        let mut injections = HashMap::new();
102        injections.insert(None, injection);
103        Self { injections }
104    }
105
106    // =========================================================================
107    // Accessors and mutators
108    // =========================================================================
109
110    /// Sets or replaces the default injection profile (all species).
111    pub fn set_default(&mut self, injection: TemporalInjection) {
112        self.injections.insert(None, injection);
113    }
114
115    /// Sets or replaces the injection profile for a named species.
116    ///
117    /// # Example
118    ///
119    /// ```rust
120    /// use chrom_rs::domain::Sample;
121    /// use chrom_rs::models::TemporalInjection;
122    ///
123    /// let mut sample = Sample::uniform(TemporalInjection::gaussian(10.0, 2.0, 0.1));
124    /// sample.set_species("B", TemporalInjection::rectangle(5.0, 15.0, 0.05));
125    /// assert!(sample.species_injection("B").is_some());
126    /// ```
127    pub fn set_species(&mut self, name: &str, injection: TemporalInjection) {
128        self.injections.insert(Some(name.to_string()), injection);
129    }
130
131    /// Returns the default injection profile, if any.
132    pub fn default_injection(&self) -> Option<&TemporalInjection> {
133        self.injections.get(&None)
134    }
135
136    /// Returns the injection profile for a named species, if any.
137    pub fn species_injection(&self, name: &str) -> Option<&TemporalInjection> {
138        self.injections.get(&Some(name.to_string()))
139    }
140
141    /// Returns `true` if no injection profile is defined.
142    pub fn is_empty(&self) -> bool {
143        self.injections.is_empty()
144    }
145
146    /// Returns a reference to the inner map, compatible with
147    /// [`PhysicalModel::set_injections`](crate::physics::PhysicalModel::set_injections).
148    pub fn as_map(&self) -> &HashMap<Option<String>, TemporalInjection> {
149        &self.injections
150    }
151}

157#[cfg(test)]
158mod tests {
159    use super::*;
160
161    #[test]
162    fn test_sample_empty() {
163        let sample = Sample::empty();
164        assert!(sample.is_empty());
165        assert!(sample.default_injection().is_none());
166    }
167
168    #[test]
169    fn test_sample_uniform() {
170        let sample = Sample::uniform(TemporalInjection::gaussian(10.0, 2.0, 0.1));
171        assert!(!sample.is_empty());
172        assert!(sample.default_injection().is_some());
173    }
174
175    #[test]
176    fn test_sample_set_default() {
177        let mut sample = Sample::empty();
178        sample.set_default(TemporalInjection::dirac(5.0, 0.1));
179        assert!(sample.default_injection().is_some());
180    }
181
182    #[test]
183    fn test_sample_set_species() {
184        let mut sample = Sample::uniform(TemporalInjection::gaussian(10.0, 2.0, 0.1));
185        sample.set_species("B", TemporalInjection::rectangle(5.0, 15.0, 0.05));
186        assert!(sample.species_injection("B").is_some());
187        assert!(sample.species_injection("A").is_none());
188    }
189
190    #[test]
191    fn test_sample_as_map_compatible_with_set_injections() {
192        let mut sample = Sample::uniform(TemporalInjection::dirac(5.0, 0.1));
193        sample.set_species("Malic", TemporalInjection::gaussian(10.0, 2.0, 0.05));
194
195        let map = sample.as_map();
196        assert!(map.contains_key(&None));
197        assert!(map.contains_key(&Some("Malic".to_string())));
198        assert_eq!(map.len(), 2);
199    }
200}
