1//! Runge-Kutta 4 (RK4) numerical solver
2//!
3//! # Mathematical Background
4//!
5//! The classical fourth-order Runge-Kutta method (RK4) is one of the most
6//! widely used numerical integrators for ordinary differential equations:
7//!
8//! ```text
9//! dy/dt = f(y, t)
10//! ```
11//!
12//! The RK4 scheme uses a weighted average of four slope estimates:
13//!
14//! ```text
15//! k₁ = f(yₙ, tₙ)
16//! k₂ = f(yₙ + dt/2 * k₁, tₙ + dt/2)
17//! k₃ = f(yₙ + dt/2 * k₂, tₙ + dt/2)
18//! k₄ = f(yₙ + dt * k₃, tₙ + dt)
19//!
20//! yₙ₊₁ = yₙ + dt/6 * (k₁ + 2k₂ + 2k₃ + k₄)
21//! ```
22//!
23//! # Characteristics
24//!
25//! - **Order**: Fourth-order accurate (error ~ O(dt⁴) per step)
26//! - **Stability**: Better stability than Euler, suitable for moderate stiffness
27//! - **Complexity**: 4 function evaluations per step
28//! - **Memory**: O(1) - stores only k₁, k₂, k₃, k₄ intermediates
29//!
30//! # Advantages
31//!
32//! ✅ Fourth-order accuracy (much better than Euler's first-order)
33//! ✅ Excellent accuracy/cost tradeoff
34//! ✅ Industry standard for non-stiff ODEs
35//! ✅ Good stability properties
36//! ✅ No tuning parameters required
37//!
38//! # Limitations
39//!
40//! ⚠️ 4× more function evaluations than Euler
41//! ⚠️ Still explicit (not ideal for very stiff problems)
42//! ⚠️ Fixed time step (not adaptive)
43//!
44//! # When to Use
45//!
46//! - **Production chromatography simulations** ✅
47//! - Non-stiff or moderately stiff problems
48//! - When accuracy is important but adaptivity is not required
49//! - Standard choice for most ODE problems
50//!
51//! # When NOT to Use
52//!
53//! - Very stiff problems → Use implicit methods (BDF, Rosenbrock)
54//! - Need error control → Use adaptive RK45 or Dormand-Prince
55//! - Extreme efficiency needed → Consider higher-order methods
56//!
57//! # Comparison with Euler
58//!
59//! | Method | Order | Evals/Step | Typical dt | Error |
60//! |--------|-------|------------|------------|-------|
61//! | Euler  | 1     | 1          | Small      | O(dt) |
62//! | RK4    | 4     | 4          | Moderate   | O(dt⁴)|
63//!
64//! **Example**: For dt = 0.01:
65//! - Euler error: ~ 0.01
66//! - RK4 error: ~ 0.00000001 (10,000× better!)
67//!
68//! # Example
69//!
70//! ```rust
71//! use chrom_rs::solver::{RK4Solver, Solver, SolverConfiguration, Scenario, DomainBoundaries};
72//! # use chrom_rs::physics::{PhysicalModel, PhysicalState, PhysicalQuantity, PhysicalData};
73//! # use nalgebra::DVector;
74//! # use serde::{Deserialize, Serialize};
75//! # #[derive(Deserialize, Serialize)]
76//! # struct MyModel;
77//! # #[typetag::serde]
78//! # impl PhysicalModel for MyModel {
79//! #     fn points(&self) -> usize { 1 }
80//! #     fn compute_physics(&self, state: &PhysicalState) -> PhysicalState { state.clone() }
81//! #     fn setup_initial_state(&self) -> PhysicalState {
82//! #         PhysicalState::new(PhysicalQuantity::Concentration, PhysicalData::Vector(DVector::from_vec(vec![1.0])))
83//! #     }
84//! #     fn name(&self) -> &str { "MyModel" }
85//! # }
86//! # fn main() -> Result<(), String> {
87//! # let model = Box::new(MyModel);
88//! # let boundaries = DomainBoundaries::temporal(model.setup_initial_state());
89//! # let scenario = Scenario::new(model, boundaries);
90//! let solver = RK4Solver::new();
91//! let config = SolverConfiguration::time_evolution(600.0, 1000);
92//!
93//! let _ = solver.solve(&scenario, &config)?;

106/// Classical fourth-order Runge-Kutta solver
107///
108/// Implements the RK4 time integration scheme with four intermediate stages
109/// per time step, providing fourth-order accuracy.
110///
111/// # Algorithm
112///
113/// For ODE system dy/dt = f(y):
114///
115/// 1. Start with initial state y₀
116/// 2. For each time step n = 0, 1, 2, ..., N-1:
117///    - **Stage 1**: k₁ = f(yₙ, tₙ)
118///      - Slope at beginning of interval
119///    - **Stage 2**: k₂ = f(yₙ + dt/2·k₁, tₙ + dt/2)
120///      - Slope at midpoint using Euler step with k₁
121///    - **Stage 3**: k₃ = f(yₙ + dt/2·k₂, tₙ + dt/2)
122///      - Slope at midpoint using Euler step with k₂
123///    - **Stage 4**: k₄ = f(yₙ + dt·k₃, tₙ + dt)
124///      - Slope at end of interval using Euler step with k₃
125///    - **Update**: yₙ₊₁ = yₙ + dt/6·(k₁ + 2k₂ + 2k₃ + k₄)
126///      - Weighted average with Simpson's rule weights
127/// 3. Return complete trajectory
128///
129/// # Stability
130///
131/// RK4 has better stability than Forward Euler. The stability region
132/// (where |R(z)| ≤ 1 for z = λ·dt) is larger, allowing bigger time steps.
133///
134/// For linear problems dy/dt = λy, RK4 is stable when:
135///
136/// ```text
137/// |1 + z + z²/2 + z³/6 + z⁴/24| ≤ 1
138/// ```
139///
140/// This allows ~2.78× larger time steps than Euler for the same stability.
141///
142/// # Error Analysis
143///
144/// - **Local truncation error**: O(dt⁵) per step
145/// - **Global error**: O(dt⁴) after T/dt steps
146/// - **Convergence**: Fourth-order convergence when refining dt
147///
148/// **Practical implication**: Halving dt reduces error by factor of 16!
149///
150/// # Example
151///
152/// ```rust
153/// # use chrom_rs::solver::{RK4Solver, Solver, SolverConfiguration, Scenario, DomainBoundaries};
154/// # use chrom_rs::physics::{PhysicalModel, PhysicalState, PhysicalQuantity, PhysicalData};
155/// # use nalgebra::DVector;
156/// # use serde::{Deserialize, Serialize};
157/// # #[derive(Deserialize, Serialize)]
158/// # struct MyModel;
159/// # #[typetag::serde]
160/// # impl PhysicalModel for MyModel {
161/// #     fn points(&self) -> usize { 1 }
162/// #     fn compute_physics(&self, state: &PhysicalState) -> PhysicalState { state.clone() }
163/// #     fn setup_initial_state(&self) -> PhysicalState {
164/// #         PhysicalState::new(PhysicalQuantity::Concentration, PhysicalData::Vector(DVector::from_vec(vec![1.0])))
165/// #     }
166/// #     fn name(&self) -> &str { "MyModel" }
167/// # }
168/// # fn main() -> Result<(), String> {
169/// # let model = Box::new(MyModel);
170/// # let boundaries = DomainBoundaries::temporal(model.setup_initial_state());
171/// # let scenario = Scenario::new(model, boundaries);
172/// let solver = RK4Solver::new();
173///
174/// // Can use larger time steps than Euler for same accuracy
175/// let config = SolverConfiguration::time_evolution(
176///     600.0,   // Total time (seconds)
177///     1000,    // Much fewer steps needed than Euler!
178/// );
179///
180/// let result = solver.solve(&scenario, &config)?;
181/// println!("Solved with {} evaluations", 4 * 1000);
182/// # Ok(())
183/// # }
184/// ```
185#[derive(Debug, Clone, Copy, Default)]

204impl Solver for RK4Solver {
205    fn solve(
206        &self,
207        scenario: &Scenario,
208        config: &SolverConfiguration,
209    ) -> Result<SimulationResult, String> {
210        // ====== Step 1: Validation ======
211
212        // Validate configuration parameters
213        config.validate()?;
214
215        // Validate scenario (model and boundaries)
216
217        scenario.validate()?;
218
219        // Forward Euler is dedicated to time evolution
220
221        let (total_time, time_steps) = match &config.solver_type {
222            SolverType::TimeEvolution {
223                total_time,
224                time_steps,
225            } => (*total_time, *time_steps),
226            other => {
227                return Err(format!(
228                    "EulerSolver only supports TimeEvolution configuration, got {}",
229                    other.name()
230                ));
231            }
232        };
233
234        // ====== Step 2: Setup ======
235
236        // Compute time step size
237        // dt = T / N where T is total time and N is number of steps
238        let dt = total_time / (time_steps as f64);
239
240        // Get initial state from scenario boundaries
241
242        let mut state = match scenario.conditions.initial_condition() {
243            Some(initial_state) => initial_state.clone(),
244            None => return Err("No initial condition found in domain boundaries".to_string()),
245        };
246
247        // preallocate storage for trajectory
248        // Reserve exact capacity to avoid reallocation during integration
249        let mut time_points = Vec::with_capacity(time_steps + 1);
250        let mut state_trajectory = Vec::with_capacity(time_steps + 1);
251
252        // Store initial condition
253
254        time_points.push(0.0);
255        state_trajectory.push(state.clone());
256
257        // ====== Step 3: Time Integration ======
258
259        // RK4 time-stepping loop
260        // For each step n = 0, 1, ..., time_steps - 1:
261        //   1. Compute k₁, k₂, k₃, k₄ (4 stages)
262        //   2. Update: yₙ₊₁ = yₙ + dt/6·(k₁ + 2k₂ + 2k₃ + k₄)
263        //   3. Store state and time point
264        //   4. Validate state
265
266        for step in 0..time_steps {
267            let t = (step as f64) * dt;
268
269            // ====== RK4 Stages ======
270
271            state.set_metadata("time".to_string(), t);
272
273            // Stage 1: Slope at beginning of interval
274            // k₁ = f(yₙ, tₙ)
275
276            let k1 = scenario.model.compute_physics(&state);
277
278            // Stage 2: Slope at midpoint using Euler prediction with k₁
279            // k₂ = f(yₙ + dt/2·k₁, tₙ + dt/2)
280
281            let state_k2 = state.clone() + k1.clone() * (dt / 2.0);
282            let k2 = scenario.model.compute_physics(&state_k2);
283
284            // Stage 3: Slope at midpoint using Euler prediction with k₂
285            // k₃ = f(yₙ + dt/2·k₂, tₙ + dt/2)
286
287            let state_k3 = state.clone() + k2.clone() * (dt / 2.0);
288            let k3 = scenario.model.compute_physics(&state_k3);
289
290            // Stage 4: Slope at end using Euler prediction with k₃
291            // k₄ = f(yₙ + dt·k₃, tₙ + dt)
292
293            let state_k4 = state.clone() + k3.clone() * dt;
294            let k4 = scenario.model.compute_physics(&state_k4);
295
296            // ====== RK4 Update ======
297
298            // Weighted combination using Simpson's rule weights:
299            // yₙ₊₁ = yₙ + dt/6·(k₁ + 2k₂ + 2k₃ + k₄)
300            //
301            // Why these weights?
302            // - k₁ and k₄ (endpoints): weight 1/6 each
303            // - k₂ and k₃ (midpoints): weight 2/6 = 1/3 each
304            // This comes from Simpson's quadrature rule for integration
305
306            let weighted_slope = k1 + k2 * 2.0 + k3 * 2.0 + k4;
307
308            state = state.clone() + weighted_slope * (dt / 6.0);
309
310            // ====== Storage ======
311
312            state_trajectory.push(state.clone());
313
314            // Store time point: t_{n+1} = (step + 1) * dt
315            // IMPORTANT: Calculate directly from index to avoid accumulation of
316            // floating-point rounding errors. This ensures that the final time point
317            // is exactly total_time (within machine epsilon).
318            //
319            // Why not t += dt? Because 0.1 is not exactly representable in binary:
320            //   0.1 ≈ 0.1000000000000000055511151231...
321            // After 100 additions: error accumulates to ~1e-14
322            // Direct calculation: (100 * 0.1) has much smaller error
323
324            time_points.push((step as f64 + 1.0) * dt);
325
326            // ====== Validation ======
327
328            validate_state(&state, step + 1)?;
329        }
330
331        // ====== Step 4: Build Result ======
332
333        let final_state = state;
334
335        // Create simulation result
336
337        let mut result = SimulationResult::new(time_points, state_trajectory, final_state);
338
339        // Add metadata
340        result.add_metadata("solver", "Runge-Kutta 4");
341        result.add_metadata("time steps", time_steps.to_string());
342        result.add_metadata("dt", dt.to_string());
343        result.add_metadata("total time", total_time.to_string());
344        result.add_metadata("function evaluations", (4 * time_steps).to_string());
345
346        Ok(result)
347    }
348
349    fn name(&self) -> &'static str {
350        "Runge Kutta (RK4)"
351    }
352}

650    fn test_rk4_convergence() {
651        let solver = RK4Solver::new();
652        let decay_rate = 0.1;
653        let total_time = 5.0;
654
655        let model = Box::new(ExponentialDecay {
656            points: 3,
657            decay_rate,
658        });
659
660        let exact_solution = (-decay_rate * total_time).exp();
661
662        // Test with different time steps
663        let vsteps = vec![50, 100, 200, 400];
664        let mut verrors: Vec<f64> = Vec::new();
665
666        for &steps in &vsteps {
667            let initial = model.setup_initial_state();
668            let boundaries = DomainBoundaries::temporal(initial);
669            let scenario = Scenario::new(
670                Box::new(ExponentialDecay {
671                    points: 3,
672                    decay_rate,

678            let result = solver.solve(&scenario, &config).unwrap();
679
680            let final_concentration = result
681                .final_state
682                .get(PhysicalQuantity::Concentration)
683                .unwrap();

685            verrors.push((final_concentration.as_vector()[0] - exact_solution).abs());
686        }

688        // Check fourth-order convergence: error(dt/2) ≈ error(dt) / 16
689

694                ratio > 12.0 && ratio < 20.0,
695                "Convergence ratio {} is not a fourth-order a step {}",
696                ratio,
697                i
698            );
699        }
