1//! Forward Euler numerical solver
2//!
3//! # Mathematical Background
4//!
5//! The Forward Euler method is the simplest explicit time-stepping scheme
6//! for solving ordinary differential equations (ODEs):
7//!
8//! $$\frac{dy}{dt} = f(y, t)$$
9//!
10//! The scheme approximates the solution at time $t_{n+1} = t_n + \Delta t$ using:
11//!
12//! $$y_{n+1} = y_n + \Delta t \cdot f(y_n, t_n)$$
13//!
14//! # Characteristics
15//!
16//! - **Order**: First-order accurate (error ~ O(dt))
17//! - **Stability**: Conditionally stable (requires small time steps)
18//! - **Complexity**: 1 function evaluation per step
19//! - **Memory**: O(1) - only stores current state
20//!
21//! # Advantages
22//!
23//! - ✅ Simplest possible time integrator
24//! - ✅ Easy to understand and implement
25//! - ✅ Low computational cost per step
26//! - ✅ Good for educational purposes
27//!
28//! # Limitations
29//!
30//! - ⚠️ First-order accuracy only (needs small dt for precision)
31//! - ⚠️ Stability restrictions (especially for stiff problems)
32//! - ⚠️ Not suitable for production chromatography simulations
33//!
34//! # When to Use
35//!
36//! - Prototyping and testing architecture
37//! - Educational demonstrations
38//! - Quick exploratory simulations
39//! - Non-stiff problems with relaxed accuracy requirements
40//!
41//! # When NOT to Use
42//!
43//! - Production chromatography simulations → Use RK4 or higher-order methods
44//! - Stiff problems → Use implicit methods
45//! - High-accuracy requirements → Use RK4, RK45, or adaptive methods
46//!
47//! # Example
48//!
49//! ```rust
50//! # use chrom_rs::solver::{EulerSolver, Solver, SolverConfiguration, Scenario, DomainBoundaries};
51//! # use chrom_rs::physics::{PhysicalModel, PhysicalState, PhysicalQuantity, PhysicalData};
52//! # use nalgebra::DVector;
53//! # struct MyModel;
54//! # impl PhysicalModel for MyModel {
55//! #     fn points(&self) -> usize { 1 }
56//! #     fn compute_physics(&self, state: &PhysicalState) -> PhysicalState { state.clone() }
57//! #     fn setup_initial_state(&self) -> PhysicalState {
58//! #         PhysicalState::new(PhysicalQuantity::Concentration, PhysicalData::Vector(DVector::from_vec(vec![1.0])))
59//! #     }
60//! #     fn name(&self) -> &str { "MyModel" }
61//! # }
62//! # fn main() -> Result<(), String> {
63//! # let model = Box::new(MyModel);
64//! # let boundaries = DomainBoundaries::temporal(model.setup_initial_state());
65//! # let scenario = Scenario::new(model, boundaries);
66//! let solver = EulerSolver::new();
67//! let config = SolverConfiguration::time_evolution(600.0, 10000);
68//!
69//! // scenario must be created with model + boundaries
70//! let result = solver.solve(&scenario, &config)?;
71//! # Ok(())
72//! # }
73//! ```

83/// Forward Euler time-stepping solver
84///
85/// Implements the simplest explicit time integration scheme:
86/// $$y_{n+1} = y_n + \Delta t \cdot f(y_n)$$
87///
88/// # Algorithm
89///
90/// For ODE system $\frac{dy}{dt} = f(y)$:
91///
92/// 1. Start with initial state $y_{0}$
93/// 2. For each time step $n \in [0, N-1]$:
94///    - Compute physics: $k = f(y_{n})$
95///    - Update state: $y_{n+1} = y_{n} + \Delta t \cdot k$
96///    - Store trajectory point
97/// 3. Return complete trajectory
98///
99/// # Stability
100///
101/// The method is **conditionally stable**. For linear problems $\frac{dy}{dt} = \lambda \cdot y$,
102/// the stability condition is:
103///
104/// $$|1 + \lambda \cdot \Delta t| \leq 1$$
105///
106/// For chromatography models, this typically requires dt to be small enough
107/// that the Courant-Friedrichs-Lewy (CFL) condition is satisfied.
108///
109/// # Error Analysis
110///
111/// - **Local truncation error**: O(dt²) per step
112/// - **Global error**: O(dt) after T/dt steps
113/// - **Convergence**: First-order convergence when refining dt
114///
115/// # Example
116///
117/// ```rust
118/// # use chrom_rs::solver::{EulerSolver, Solver, Scenario, DomainBoundaries, SolverConfiguration};
119/// # use chrom_rs::physics::{PhysicalModel, PhysicalState, PhysicalQuantity, PhysicalData};
120/// # use nalgebra::DVector;
121/// # struct MyModel;
122/// # impl PhysicalModel for MyModel {
123/// #     fn points(&self) -> usize { 1 }
124/// #     fn compute_physics(&self, state: &PhysicalState) -> PhysicalState { state.clone() }
125/// #     fn setup_initial_state(&self) -> PhysicalState {
126/// #         PhysicalState::new(PhysicalQuantity::Concentration, PhysicalData::Vector(DVector::from_vec(vec![1.0])))
127/// #     }
128/// #     fn name(&self) -> &str { "MyModel" }
129/// # }
130/// # fn main() -> Result<(), String> {
131/// # let model = Box::new(MyModel);
132/// # let boundaries = DomainBoundaries::temporal(model.setup_initial_state());
133/// # let scenario = Scenario::new(model, boundaries);
134/// # let config = SolverConfiguration::time_evolution(1.0, 10);
135/// let solver = EulerSolver::new();
136/// let result = solver.solve(&scenario, &config)?;
137/// # Ok(())
138/// # }
139/// ```

159impl Solver for EulerSolver {
160    fn solve(
161        &self,
162        scenario: &Scenario,
163        config: &SolverConfiguration,
164    ) -> Result<SimulationResult, String> {
165        // ====== Step 1: Validation ======
166
167        // Validate configuration parameters
168        config.validate()?;
169
170        // Validate scenario (model and boundaries)
171
172        scenario.validate()?;
173
174        // Forward Euler is dedicated to time evolution
175
176        let (total_time, time_steps) = match &config.solver_type {
177            SolverType::TimeEvolution {
178                total_time,
179                time_steps,
180            } => (*total_time, *time_steps),
181            other => {
182                return Err(format!(
183                    "EulerSolver only supports TimeEvolution configuration, got {}",
184                    other.name()
185                ));
186            }
187        };
188
189        // ====== Step 2: Setup ======
190
191        // Compute time step size
192        // dt = T / N where T is total time and N is number of steps
193        let dt = total_time / (time_steps as f64);
194
195        // Get initial state from scenario boundaries
196
197        let mut state = match scenario.conditions.initial_condition() {
198            Some(initial_state) => initial_state.clone(),
199            None => return Err("No initial condition found in domain boundaries".to_string()),
200        };
201
202        // preallocate storage for trajectory
203        // Reserve exact capacity to avoid reallocation during integration
204        let mut time_points = Vec::with_capacity(time_steps + 1);
205        let mut state_trajectory = Vec::with_capacity(time_steps + 1);
206
207        // Store initial condition
208
209        time_points.push(0.0);
210        state_trajectory.push(state.clone());
211
212        // ====== Step 3: Time Integration ======
213
214        // Forward Euler time-stepping loop
215        // For each step n = 0, 1, ..., time_steps - 1:
216        //   1. Compute f(y_n, t_n) using the physical model
217        //   2. Update: y_{n+1} = y_n + dt * f(y_n, t_n)
218        //   3. Store state and time point
219        //   4. Validate state (check for NaN, physical bounds, etc.)
220
221        for step in 0..time_steps {
222            // Current time
223            let t = dt * step as f64;
224
225            // ====== Euler step ======
226
227            // 0. Add metadata time to be used by injection condition
228
229            state.set_metadata("time".to_string(), t);
230
231            // 1. Compute physics: f(y_n, t_n)
232            //    This returns the right-hand side of dy/dt = f(y, t)
233            //    For chromatography: transport + dispersion + adsorption terms
234            let physics: PhysicalState = scenario.model.compute_physics(&state);
235
236            // 2. Update state: y_{n+1} = y_n + dt * f(y_n)
237            //    PhysicalState implements Add and Mul<f64>, so this works naturally
238            //    The clone() is necessary because Add consumes self
239            state = state.clone() + physics * dt;
240
241            // ====== Storage ======
242
243            // Store new state in trajectory
244            // This creates a full history of the simulation for later analysis
245
246            state_trajectory.push(state.clone());
247
248            // Store time point: t_{n+1} = (step + 1) * dt
249            // IMPORTANT: Calculate directly from index to avoid accumulation of
250            // floating-point rounding errors. This ensures that the final time point
251            // is exactly total_time (within machine epsilon).
252            //
253            // Why not t += dt? Because 0.1 is not exactly representable in binary:
254            //   0.1 ≈ 0.1000000000000000055511151231...
255            // After 100 additions: error accumulates to ~1e-14
256            // Direct calculation: (100 * 0.1) has much smaller error
257            time_points.push((step as f64 + 1.0) * dt);
258
259            // ====== Validation ======
260
261            // Check for numerical issues (NaN, Inf, etc.)
262            // This helps catch problems early rather than propagating them
263            solver::validate_state(&state, step + 1)?;
264        }
265
266        // ====== Step 4: Build Result ======
267
268        // Extract final state (as the last element of trajectory)
269        let final_state: PhysicalState = state;
270
271        // Create a simulation result
272
273        let mut result = SimulationResult::new(time_points, state_trajectory, final_state);
274
275        // Add metadata for diagnostics and reproducibility
276
277        result.add_metadata("solver", "Forward Euler");
278        result.add_metadata("time steps", time_steps.to_string());
279        result.add_metadata("dt", dt.to_string());
280        result.add_metadata("total time", total_time.to_string());
281
282        Ok(result)
283    }
284
285    fn name(&self) -> &'static str {
286        "Forward Euler"
287    }
288}

331            PhysicalState::new(
332                PhysicalQuantity::Concentration,

338            "Exponential Decay"

531    fn test_euler_convergence() {
532        // Tests that error decrease linearly with dt (first-order convergence)
533
534        let solver = EulerSolver::new();
535        let decay_rate = 0.5;
536        let total_time = 5.0;
537
538        let model = Box::new(ExponentialDecay {
539            points: 3,
540            decay_rate,
541        });
542
543        let exact = (-decay_rate * total_time).exp();
544
545        // Running tests with different timesteps
546
547        let vsteps: Vec<usize> = vec![100, 200, 400, 800];
548        let mut verrors: Vec<f64> = Vec::new();
549
550        for &steps in &vsteps {
551            let initial = model.setup_initial_state();
552            let boundaries = DomainBoundaries::temporal(initial);
553            let scenario = Scenario::new(
554                Box::new(ExponentialDecay {
555                    points: 3,
556                    decay_rate,
557                }),
558                boundaries,
559            );
560
561            let config = SolverConfiguration::time_evolution(total_time, steps);
562            let result = solver.solve(&scenario, &config).unwrap();
563
564            let final_state = result
565                .final_state
566                .get(PhysicalQuantity::Concentration)
567                .unwrap();
568            let actual = final_state.as_vector()[0];
569            let error = (actual - exact).abs();
570            verrors.push(error);
571        }
572
573        // Check first-order convergence: error(dt/2) ≈ error(dt) / 2
574
575        for i in 0..verrors.len() - 1 {
576            let ratio = verrors[i] / verrors[i + 1];
577
578            assert!(
579                ratio > 1.8 && ratio < 2.2,
580                "Convergence ration {} not a first order at step {}",
581                ratio,
582                i
583            );
584        }

662    fn test_euler_time_precision() {
663        // Specific test for floating-point precision in time points
664        // This test would fail with accumulation (t += dt)
665
666        let solver = EulerSolver::new();
667
668        let model = Box::new(ConstantGrowth {
669            points: 3,
670            growth_rate: 1.0,
671        });
672
673        let initial = model.setup_initial_state();
674        let boundaries = DomainBoundaries::temporal(initial);
675        let scenario = Scenario::new(model, boundaries);
676
677        // Use a total_time that's not exactly representable in binary
678        // and a number of steps that will magnify rounding errors
679        let total_time = 10.0;
680        let time_steps = 100;
681        let config = SolverConfiguration::time_evolution(total_time, time_steps);
682
683        let result = solver.solve(&scenario, &config).unwrap();
684
685        // With direct calculation: (step + 1) * dt
686        // Final time should be exactly total_time within machine epsilon (~2e-16)
687        let final_time = *result.time_points.last().unwrap();
688
689        // Machine epsilon for f64 is ~2.22e-16
690        // With direct calculation, error should be O(epsilon)
691        // With accumulation, error would be O(n * epsilon) ≈ 100 * 2e-16 = 2e-14
692        assert!(
693            (final_time - total_time).abs() < 1e-14,
694            "Direct calculation maintains precision: {} ≈ {} (error: {:e})",
695            final_time,
696            total_time,
697            (final_time - total_time).abs()
698        );

904            }
905
906            fn name(&self) -> &str {
907                "Multiple Physical Quantity Model"
908            }
909        }
910
