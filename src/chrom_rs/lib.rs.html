1//! chrom-rs: Chromatography Simulation Framework
2//!
3//! A flexible and extensible framework for simulating chromatographic processes
4//! using numerical methods. Built with Rust for performance and safety.
5//!
6//! # Architecture
7//!
8//! chrom-rs is built on two core principles:
9//!
10//! 1. **Separation of Physics and Numerics**
11//!    - Physical models define equations (what to solve)
12//!    - Numerical solvers provide methods (how to solve)
13//!
14//! 2. **Extensibility and Type Safety**
15//!    - Trait-based design for easy extension
16//!    - Type-safe state management
17//!    - Stable API (v0.1.0+)
18//!
19//! # Quick Start
20//!
21//! ```rust
22//! use chrom_rs::physics::{PhysicalModel, PhysicalState, PhysicalQuantity, PhysicalData};
23//! use chrom_rs::solver::{EulerSolver, Solver, SolverConfiguration, Scenario, DomainBoundaries};
24//! use nalgebra::DVector;
25//!
26//! # struct MyModel;
27//! # impl PhysicalModel for MyModel {
28//! #     fn points(&self) -> usize { 1 }
29//! #     fn compute_physics(&self, state: &PhysicalState) -> PhysicalState { state.clone() }
30//! #     fn setup_initial_state(&self) -> PhysicalState {
31//! #         PhysicalState::new(PhysicalQuantity::Concentration, PhysicalData::Vector(nalgebra::DVector::from_vec(vec![1.0])))
32//! #     }
33//! #     fn name(&self) -> &str { "MyModel" }
34//! # }
35//! # fn main() -> Result<(), String> {
36//! // 1. Configure physical model and scenario
37//! let model = Box::new(MyModel);
38//! let initial_state = model.setup_initial_state();
39//! let boundaries = DomainBoundaries::temporal(initial_state);
40//! let scenario = Scenario::new(model, boundaries);
41//!
42//! // 2. Configure solver
43//! let config = SolverConfiguration::time_evolution(
44//!     600.0,    // 10 minutes total time
45//!     1000,     // 1000 time steps
46//! );
47//!
48//! // 3. Run simulation
49//! let solver = EulerSolver::new();
50//! let result = solver.solve(&scenario, &config)?;
51//!
52//! // 4. Access results
53//! println!("Simulation completed!");
54//! println!("Trajectory length: {}", result.len());
55//! # Ok(())
56//! # }
57//! ```
58//!
59//! # Modules

61//! - [`physics`]: Physical models (equations)
62//! - [`solver`]: Numerical solvers (methods)
63//! - `cli`: Command-line interface (optional)
64//! - [`output`]: Result visualization and export (publishing)
