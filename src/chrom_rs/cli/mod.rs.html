1//! Command-line interface for `chrom-rs`.
2//!
3//! Assembles the `dynamic-cli` application from a declarative YAML
4//! configuration embedded at compile time and wires it to the simulation
5//! pipeline defined in the [`app`](crate::cli::app) module.
6//!
7//! # Entry point
8//!
9//! ```rust,no_run
10//! chrom_rs::cli::build_app()
11//!     .expect("CLI initialisation failed")
12//!     .run();
13//! ```
14//!
15//! # Command surface (v0.2.0)
16//!
17//! ```text
18//! chrom-rs run [--project-dir <dir>]
19//!              --model    <file.yml>
20//!              --scenario <file.yml>
21//!              --solver   <file.yml>
22//!              [--output-csv   <file.csv>]
23//!              [--output-plot  <file.png>]
24//!              [--export-json  <file.json>]
25//! ```
26
27/// Execution context, command handlers, and simulation helpers.
28///
29/// All runtime state ([`ChromContext`](crate::cli::app::ChromContext)),
30/// input validation, and the `run` command handler
31/// ([`RunHandler`](crate::cli::app::RunHandler)) live here.
32pub mod app;
33
34use anyhow::anyhow;
35use dynamic_cli::config::loader::load_yaml;
36use dynamic_cli::{CliApp, CliBuilder};
37
38use app::{ChromContext, RunHandler};
39
40// ============================================================================
41// Embedded command configuration
42// ============================================================================
43
44/// YAML command configuration, embedded at compile time from
45/// `src/cli/commands.yml`.
46///
47/// Parsed once in [`build_app`] via `load_yaml`. Keeping the declarations in
48/// YAML lets maintainers adjust help text, aliases, and option metadata
49/// without touching Rust code.
50const COMMANDS_YML: &str = include_str!("commands.yml");
51
52/// Handler name that must match the `implementation:` field of the `run`
53/// command in `commands.yml`.
54const RUN_HANDLER_NAME: &str = "run_handler";
55
56// ============================================================================
57// build_app
58// ============================================================================
59
60/// Assembles and returns the fully configured [`CliApp`].
61///
62/// Parses the embedded command YAML, wires
63/// [`RunHandler`] and a fresh
64/// [`ChromContext`], then delegates to
65/// `CliBuilder::build`.
66///
67/// # Errors
68///
69/// - The embedded YAML is malformed (compile-time regression).
70/// - The builder detects a missing required handler.
71pub fn build_app() -> anyhow::Result<CliApp> {
72    let config =
73        load_yaml(COMMANDS_YML).map_err(|e| anyhow!("embedded commands.yml is invalid: {e}"))?;
74
75    CliBuilder::new()
76        .config(config)
77        .context(Box::new(ChromContext::new()))
78        .register_handler(RUN_HANDLER_NAME, Box::new(RunHandler))
79        .build()
80        .map_err(|e| anyhow!("CLI builder error: {e}"))
81}

91    #[test]

97    fn test_commands_yml_is_valid_yaml() {
98        use dynamic_cli::config::loader::load_yaml;
99        let config = load_yaml(COMMANDS_YML).expect("COMMANDS_YML must be valid");
100        assert!(config.commands.iter().any(|c| c.name == "run"));
