1//! Loader for `model.yml` / `model.json`.
2//!
3//! `model.yml` describes the physical column and its parameters.
4//! The injection profile is intentionally absent (or set to `None`) here —
5//! it is supplied by `scenario.yml` and applied by [`scenario::load_scenario`].
6//!
7//! # File format
8//!
9//! The file is deserialised via `typetag`, which uses the type name as the
10//! root key. Supported formats: `.yml` / `.yaml` (serde_yaml) and `.json`
11//! (serde_json). The extension determines the parser.
12//!
13//! ## `LangmuirSingle` example
14//!
15//! ```yaml
16//! LangmuirSingle:
17//!   lambda: 1.2
18//!   langmuir_k: 0.4
19//!   port_number: 2.0
20//!   porosity: 0.4
21//!   velocity: 0.001
22//!   column_length: 0.25
23//!   nz: 100
24//!   injection:
25//!     type: None
26//! ```
27//!
28//! ## `LangmuirMulti` example
29//!
30//! ```yaml
31//! LangmuirMulti:
32//!   species:
33//!     - name: Ascorbic
34//!       lambda: 1.0
35//!       langmuir_k: 1.1
36//!       port_number: 2
37//!       injection:
38//!         type: None
39//!     - name: Erythorbic
40//!       lambda: 1.0
41//!       langmuir_k: 1.7
42//!       port_number: 2
43//!       injection:
44//!         type: None
45//!   n_points: 100
46//!   porosity: 0.4
47//!   velocity: 0.001
48//!   column_length: 0.25
49//! ```
50//!
51//! The `injection: type: None` placeholder is replaced by
52//! [`scenario::load_scenario`] after the model is loaded.
53
54use crate::config::ConfigError;
55use crate::physics::PhysicalModel;

57/// Loads a physical model from a configuration file.
58///
59/// The file is deserialised via `typetag` — the root key must match the
60/// registered type name (e.g. `LangmuirSingle`, `LangmuirMulti`).
61///
62/// The returned model has its injection profile set to `None` by convention.
63/// Call [`scenario::load_scenario`](super::scenario::load_scenario) to apply
64/// the injection defined in `scenario.yml`.
65///
66/// # Errors
67///
68/// - [`ConfigError::Io`] if the file cannot be read.
69/// - [`ConfigError::UnsupportedFormat`] if the extension is not `.yml`,
70///   `.yaml`, or `.json`.
71/// - [`ConfigError::Parse`] if the file content is not valid YAML/JSON or
72///   does not match a known `PhysicalModel` type.
73///
74/// # Example
75///
76/// ```rust,no_run
77/// use chrom_rs::config::model::load_model;
78///
79/// let model = load_model("model_tfa.yml").unwrap();
80/// println!("Loaded: {}", model.name());
81/// ```
82pub fn load_model(path: &str) -> Result<Box<dyn PhysicalModel>, ConfigError> {
83    crate::config::load_from_file::<Box<dyn PhysicalModel>>(path)
84}

97    /// Writes a temporary YAML file and returns it (kept alive for the test).
98    fn tmp_yaml(content: &str) -> tempfile::NamedTempFile {
99        let mut f = tempfile::Builder::new().suffix(".yml").tempfile().unwrap();
100        write!(f, "{content}").unwrap();
101        f
102    }
103
104    /// Writes a temporary JSON file and returns it (kept alive for the test).
105    fn tmp_json(content: &str) -> tempfile::NamedTempFile {
106        let mut f = tempfile::Builder::new().suffix(".json").tempfile().unwrap();
107        write!(f, "{content}").unwrap();
108        f
109    }

152    #[test]
153    fn test_load_langmuir_single_yaml() {
154        let f = tmp_yaml(LANGMUIR_SINGLE_YAML);
155        let model = load_model(f.path().to_str().unwrap())
156            .expect("load_model must succeed for LangmuirSingle YAML");
157
158        assert_eq!(
159            model.name(),
160            "Langmuir single specie with temporal injection"
161        );
162        assert_eq!(model.points(), 100);
163    }

169            .expect("load_model must succeed for LangmuirMulti YAML");
170
171        assert_eq!(model.name(), "Langmuir Multi-Species");
172        assert_eq!(model.points(), 50);
173    }
